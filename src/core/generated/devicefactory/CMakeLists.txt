function(forte_register_device class)
    file(REAL_PATH ${class}.h header BASE_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
    set_property(TARGET forte APPEND PROPERTY forte_device "${header}://:${class}")
endfunction()

set(FORTE_DEVICE_DEFAULT "RMT_DEV" CACHE STRING "The default device to use")

function(_forte_generate_core_devicefactory)
	get_property(registered TARGET forte PROPERTY forte_device)
	set(default ${FORTE_DEVICE_DEFAULT})

    set(items)
    set(includes "")
    set(entries "")
    foreach (reg IN LISTS registered)
        string(REGEX REPLACE "(.*)://:(.*)" "\\1" header "${reg}")
        string(REGEX REPLACE "(.*)://:(.*)" "\\2" class "${reg}")

        list(APPEND items ${class})
        string(APPEND entries "ENTRY(${class})\n")
        string(APPEND includes "#include \"${header}\"\n")
    endforeach()

	if (NOT "${default}" IN_LIST items)
		message(FATAL_ERROR "${default} is not a valid device. Choose one of the following: ${items}")
	endif()

	list(LENGTH registered len)
	if (len GREATER 1)
		set(multiple 1)
	else()
		set(multiple 0)
	endif()
    message(STATUS "Device Classes: ${items}")
	configure_file(${CMAKE_CURRENT_FUNCTION_LIST_DIR}/devicefactory.h.in "${FORTE_GENERATED_FILES_DIR}/devicefactory.h")
	configure_file(${CMAKE_CURRENT_FUNCTION_LIST_DIR}/devicefactory.cpp.in "${FORTE_GENERATED_FILES_DIR}/devicefactory.cpp")
    target_sources(forte-core PRIVATE ${FORTE_GENERATED_FILES_DIR}/devicefactory.cpp)
endfunction()

cmake_language(DEFER DIRECTORY ../../.. CALL _forte_generate_core_devicefactory)
