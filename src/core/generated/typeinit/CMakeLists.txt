function(_forte_generate_typeinit)
    forte_find_sources_recursive(source_files forte)

    set(dtstatus)
    set(fbstatus)
    set(includes)
    set(entries)
    foreach (source IN LISTS source_files)
        if (source MATCHES "/typelib.h\$")
            continue()
        endif()
        file(STRINGS "${source}" definitions REGEX "DEFINE_(FIRMWARE_DATATYPE|FIRMWARE_FB|GENERIC_FIRMWARE_FB|ADAPTER_TYPE)\\([a-zA-Z_0-9]+")
        cmake_path(REPLACE_EXTENSION source .h OUTPUT_VARIABLE sourceh)

        foreach (def IN LISTS definitions)
            string(REGEX REPLACE "(DEFINE_|\\(.*)" "" deftype "${def}")
            string(REGEX REPLACE "DEFINE_[a-zA-Z_0-9]*\\(([a-zA-Z_0-9]*)[^a-zA-Z_0-9].*" "\\1" class "${def}")
            string(REGEX REPLACE "DEFINE_[a-zA-Z_0-9]*\\([a-zA-Z_0-9]*[^a-zA-Z_0-9].*STRID\\((.*)\\)\\)" "\\1" name "${def}")

            list(APPEND includes "#include \"${sourceh}\"\n")
            if (deftype STREQUAL "FIRMWARE_DATATYPE")
                list(APPEND dtstatus "${name}")
                list(APPEND entries "ENTRY(CIEC_${class})\n")
            else()
                list(APPEND fbstatus "${name}")
                list(APPEND entries "ENTRY(${class})\n")
            endif()
        endforeach()
    endforeach()

    list(SORT dtstatus)
    list(SORT fbstatus)
    list(SORT includes)
    list(SORT entries)
    list(REMOVE_DUPLICATES dtstatus)
    list(REMOVE_DUPLICATES fbstatus)
    list(REMOVE_DUPLICATES includes)
    list(REMOVE_DUPLICATES entries)
    string(JOIN " " dtstatus ${dtstatus})
    string(JOIN " " fbstatus ${fbstatus})
    string(JOIN ";\n" includes ${includes})
    string(JOIN "\n" entries ${initfunctions})

    message(STATUS "FB/Adapter Types: ${fbstatus}")
    message(STATUS "Data Types: ${dtstatus}")
    configure_file("${CMAKE_CURRENT_FUNCTION_LIST_DIR}/typeinit.cpp.in" "${FORTE_GENERATED_FILES_DIR}/typeinit.cpp")
    target_sources(forte-core PRIVATE "${FORTE_GENERATED_FILES_DIR}/typeinit.cpp")
endfunction()

cmake_language(DEFER DIRECTORY ../../.. CALL cmake_language DEFER DIRECTORY . CALL _forte_generate_typeinit)
